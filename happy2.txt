    <h1>✨ 我的周邊收集整理櫃 ✨</h1>

    <div class="grid-container" id="grid">
        </div>

    <script>
        // 1. 【在這裡修改你的周邊資料】
        // id 必須唯一；img 可以放網路圖片網址，或是放在同個資料夾內的圖片路徑（例如 'images/p1.jpg'）
        const itemsData = [
            { id: 'item_1', title: '人生四格', desc: '人生四格小卡', img: 'stand.jpg' },
            { id: 'item_2', title: '角色壓克力立牌', desc: '夏日祭典主題立牌', img: 'card.jpg' },
            { id: 'item_3', title: '限定珍藏同人誌', desc: '全彩 40P 特典版', img: 'stand.jpg' }
        ];

        const gridContainer = document.getElementById('grid');

        // 2. 渲染畫面與綁定事件
        itemsData.forEach(item => {
            // 從 LocalStorage 讀取歷史數量，如果沒有就預設為 0
            const savedCount = localStorage.getItem(item.id) || 0;

            // 建立卡片 HTML
            const card = document.createElement('div');
            card.className = 'card';
            card.innerHTML = `
                <div class="img-container">
                    <img src="${item.img}" alt="${item.title}">
                </div>
                <div class="content">
                    <div>
                        <div class="title">${item.title}</div>
                        <div class="description">${item.desc}</div>
                    </div>
                    <div class="counter-container">
                        <button class="btn" onclick="updateCount('${item.id}', -1)">-</button>
                        <span class="count" id="count-${item.id}">${savedCount}</span>
                        <button class="btn" onclick="updateCount('${item.id}', 1)">+</button>
                    </div>
                </div>
            `;
            gridContainer.appendChild(card);
        });

        // 3. 加減數量與儲存功能
        function updateCount(id, change) {
            const countSpan = document.getElementById(`count-${id}`);
            let currentCount = parseInt(countSpan.innerText);
            
            // 計算新數量，並確保不會變成負數
            let newCount = currentCount + change;
            if (newCount < 0) newCount = 0;

            // 更新畫面
            countSpan.innerText = newCount;
            // 儲存到瀏覽器的 LocalStorage
            localStorage.setItem(id, newCount);
        }
    </script>
